MySQL操作基础

数据库连接

# 通过命令行方式连接MySQL数据库
mysql -h localhost -P 3306 -u root -p

# -h:指定要连接的数据库，通常为IP地址
# -P:指定连接MySQL数据库的端口（默认为3306）
# -u:指定连接数据库的用户
# -p:指定连接数据库用户的密码

数据库备份

# 备份数据库中所有表结构和数据
mysqldump -u root -p test_db > test_db_backup.sql

数据库还原

# 还原数据库表到指定数据库
mysql -u root -p test_new_db < test_bd_backup.sql

与数据库相关的操作

创建数据库

# 最简单的创建MySQL数据库的语句(其他参数会使用默认参数)
create database test_db;

# 创建数据库（屏蔽报错）
create database if not exists test_db;

# 自定义数据库的参数
create database if not exists test_db
default character set utf8mb4
default collate utf8mb4_general_ci;

修改数据库

# 修改指定数据库的字符集和校验集
alter database [数据库名]{
[default] character set <字符集名> | 
[default] collate <校对规则名>
}

-- 修改数据库的字符集为utf8mb4和校对规则名为utf8mb4_general_ci
alter database test_db
default character set utf8mb4
default collate utf8mb4_general_ci;

删除数据库

# 删除数据库（当指定数据库不存在时会报错）
drop database test_db;

# 删除数据库（先判断数据库是否存在）
drop database if exists test_db;

使用数据库

​ 在最初连接数据库时，若没有指定选择的数据库参数，就没有任何数据库打开供你使用。因此在你执行任意数据库操作之前需要选择一个数据库。

# 选择要使用的demo数据库
use test_db;

查看数据库的字符集

# 查看当前数据库使用的字符集
show character set;

查看数据库的排序和校对规则

# 查看当前数据库的排序和校对规则
show collation;

查看数据库

# 显示DBMS中所有的数据库
show databases;

# 显示与条件匹配的数据库
show databases like '%_db'        -- 以_db结尾的数据库
show databases like 'user%'        -- 以user开头的数据库
show databases like ‘%user%'    -- 名称中包含user的数据库

显示所有数据表

# 显示当前数据库中所有数据表
show tables;

显示服务器状态信息

# 显示服务器状态信息
show status;

显示服务器错误信息

# 显示服务器错误信息

显示服务器警告信息

# 显示服务器警告信息

显示权限信息

与数据表相关的操作

创建数据表

# 创建功能较齐全的数据表
create table userinfo(
id int(5) zerofill unsigned not null primary key auto_increment,
username varchar(8) not null unique,
password varchar(16) not null,
gender enum('m','w') not null default 'w');
# zerofill:0填充
# unsigned:无符号
# no null:不能为空
# primary keyL主建
# auto_increment:自增长
# unique:唯一键
# default:默认值

-- 不设置表的字符集时，创建的表默认继承数据库的字符集

修改数据表

# 修改表名
ALTER TABLE userinfo RENAME TO user_info;
-- 语法规则
ALTER TABLE <旧表名> RENAME [TO] <新表名>;

# 修改表字符集
ALTER TABLE userinfo DEFAULT CHARACTER SET utf8mb4 DEFAULT COLLATE utf8mb4_general_ci;
-- 语法规则
ALTER TABLE 表名 [DEFAULT] CHARACTER SET <字符集名> [DEFAULT] COLLATE <校对规则名>;

# 添加字段
# 在末尾添加字段
ALTER TABLE userinfo ADD age INT NULL;
-- 语法规则
ALTER TABLE <表名> ADD <新字段名><数据类型>[约束条件];

# 在开头添加字段
ALTER TABLE userinfo ADD age INT NULL FIRST;
-- 语法规则
ALTER TABLE <表名> ADD <新字段名> <数据类型> [约束条件] FIRST;

# 在中间添加字段
ALTER TABLE userinfo ADD age INT NULL AFTER username;
-- 语法规则
ALTER TABLE <表名> ADD <新字段名> <数据类型> [约束条件] AFTER <已经存在的字段名>;

# 修改字段名称及数据类型
ALTER TABLE userinfo CHANGE PASSWORD passwd VARCHAR ( 16 );
-- 语法规则
ALTER TABLE <表名> CHANGE <旧字段名> <新字段名> <新数据类型>；

# 修改列数据
ALTER TABLE userinfo MODIFY passwd VARCHAR ( 8 );
-- 语法规则
ALTER TABLE <表名> MODIFY <字段名> <数据类型>

# 删除字段
ALTER TABLE userinfo DROP ( passwd );
-- 语法规则
ALTER TABLE <表名> DROP <字段名>；

删除数据表

# 删除指定数据表
drop table user_info;
-- 语法规则
DROP TABLE [IF EXISTS] 表名1 [ ,表名2, 表名3 ...]

增加

向userinfo数据库中添加数据。

# 标准添加数据的方式
INSERT INTO userinfo ( username, PASSWORD, gender )
VALUES
    ( 'admin', '123456', 'w' );
-- 不需要指定主键，因为主建设置了自增长

# 缺省值添加
INSERT INTO userinfo
VALUES
    (2, 'user', '123456', 'w' );
-- 使用缺省方式时即使主建设置了自增长，但在添加数据时也必须设置主键值

# 一次性添加多行数据
INSERT INTO userinfo ( username, PASSWORD, gender )
VALUES
    ( 'Alice', '123456', 'w' ),
    ( 'Bob', '123456', 'm' ),
    ( 'root', '123456', 'w' ),
    ( 'usr', '123456', 'w' ),
    ( 'client', '123456', 'w' );

# 使用select into ... from一次性添加多行数据（复制查询结果到另一张表中）
INSERT INTO user_info ( id, username, password, gender ) SELECT
* 
FROM
    userinfo;

更新

-- 语法规则
UPDATE <表名> SET 字段 1=值 1 [,字段 2=值 2… ] [WHERE 子句 ]
[ORDER BY 子句] [LIMIT 子句]

# 修改一个字段值
UPDATE userinfo 
SET username = 'test' 
WHERE
    username = 'root';

# 修改多个字段值
UPDATE userinfo 
SET username = 'test',
passwd = 'test';
WHERE
    username = 'root';

# 综合条件限定修改的字段值
UPDATE user_info 
SET username = 'test',
PASSWORD = 'test' 
WHERE
    gender = 'w' 
ORDER BY
    id 
    LIMIT 2
-- limit 用于限定被修改的行数

-- 注意
1. 更新表中的字段值时一定要加上限制条件(where limit)

查询

基本查询

# 检索单个列
# 显示userinfo表中username项的所有信息
select username from userinfo;

# 检索多列
# 显示userinfo表中username和passwd项的所有信息
select username,passwd from userinfo;

# 检索所有列
# 显示userinfo表中所有列的信息
select * from userinfo;

# 所有完全限定的表名
select userinfo.username from userinfo;
# 单表查询时可省略，多表查询时有奇效

条件查询数据

# 单一条件查询
select * from userinfo where gender = 'w';

# 多条件查询
# 逻辑与(and &&)
select * from userinfo where gender = 'w' and username = 'Bob';
select * from userinfo where gender = 'w' && username = 'Bob';
# 逻辑或(or ||)
select * from userinfo where gender = 'w' or username = 'Bob';
select * from userinfo where gender = 'w' || username = 'Bob';
# 逻辑异或
select * from userinfo where gender = 'w' xor username = 'Bob';

模糊查询

# 精确查询
select * from userinfo where username like 'oo';

# 多字符通配符
# 前缀查询
select * from userinfo where username like 'r%';

# 后缀查询
select * from userinfo where username like '%t';

# 关键字查询
select * from userinfo where username like '%oo%';

# 单字符通配符
# 以r开头共四个字符
select * from userinfo where username like 'r___';

# 以t结尾共四个字符
select * from userinfo where username like '___t';

# 中间为oo共四个字符
select * from userinfo where username like '_oo_';

# 区分大小写进行模糊查询
select * from userinfo where username like binary 'r%';

# 匹配以%号结尾的用户名
select * from userinfo where username like '%\%';

检索数据唯一性（去重）

# 检索不同的行
# 显示的结果中不存在重复行
select distinct username,passwd from userinfo; 
# distinct应用于所有列

限制数据行数

# 显示结果显示的行数
# 显示查询中0行开始的5行数据
select username from userinfo limit 0,5;
# limit m,n  =  limit n offset m(5.0开始)
#  第一个参数为开始位置（可以不指定，若不指定则从头开始）
#  第二个参数为要检索的行数（必须指定）

排序数据

# 单字段排序检索数据
# 显示username和passwd并按username中字母顺序排序数据
select username,passwd from userinfo order by username;

# 多多字段排序检索数据
# 首先按id排序,然后再按username排序
select id,username,passwd from userinfo order by id,username;

# 指定排序方向
# 按username降序排序
select username,passwd from userinfo order by username desc;
# 若降序则添加desc，若升序则添加asc或不指定

# 混杂
# 将最贵的显示在最上面且只显示前20条数据
select prod_id,prod_price,prod_name
from products
order by prod_price desc,prod_name
limit 0,20;

过滤数据

# 检查单个值
select prod_name,prod_price 
from products
where prod_name = 'fuses';

select prod_name,prod_price 
from products
where prod_price between 5 and 10;
# 条件：
# = :等于
# <>:不等于
# !=:不等于
# > :大于
# < :小于
# >=:大于等于
# <=:小于等于
# between:在指定的两个值之间

# in操作符
select prod_name,prod_price
from products
where prod_id in (1002,1003);

# 检查多个值
select prod_id,prod_name,prod_price
from products
where prod_name = 'juese' and prod_price > '5';
# 逻辑符号：
# and:逻辑与
# or :逻辑或
# xor:逻辑异或

# 空值检查
select prod_name
from products
where prod_prices is Null

# not操作符
select prod_name
from products
where prod_id not in (1002,1003)
order by prod_name;

# 通配符
#     百分号
select prod_id,prod_name,prod_price
from products
where prod_name like '%juese%';
#    下划线
select prod_id,prod_name,prod_price
from produces
where prod_name like '_uese';

# 正则
select prod_id,prod_name,prod_price
from products
where prod_name regexp 'juese';
select prod_id,prod_name,prod_price
from products
where prod_name regexp '.uese';
select prod_id,prod_name,prod_price
from products
where prod_name regexp '[jkl]uese';
select prod_id,prod_name,prod_price
from products
where prod_name regexp 'juese|apple';
select prod_id,prod_name,prod_price
from products
where prod_name regexp '[a-z]uese';

字符类

类 说明
[:alnum:] 任意字母和数字（同[a-zA-Z0-9]）
[:alpha:] 任意字母（同[a-zA-Z]）
[:lower:] 任意小写字母（同[a-z]）
[:upper:] 任意大写字母（同[A-Z]）
[:digit:] 任意数字（同[0-9]）
[:xdigit:] 任意十六进制数字（同[a-fA-F0-9]）
[:graph:] 任意可打印字符，但不包含空格
[:print:] 任意可打印字符
[:blank:] 空格和制表（同[\\t]）
[:space:] 包括空格在内的任意空白字符（同[\\f\\n\\r\\t\\v]）
[:cntrl:] ASCII控制字符（ASCII0到31和127）
[:punct:] 既不在[:alnum:]又不在[:cntrl:]中的任意字符

定位

元字符 说明
^ 文本的开始
$ 文本的结束
[[:<:]] 词的开始
[[:>:]] 词的结束

分组查询

# 单个字段进行分组
-- 男女各有多少人
select gender,count(gender) 个数 from userinfo group by gender;
-- 聚合函数
- count()    :计次
- sum()        :求和
- avg()        :求平均值
- max()        :求最大值
- min()        :求最小值

# 多个字段进行分组
-- 各班级中男女各有多少人
select classis,gender,count(gender) from userinfo group by classid,gender;

# 男女生各有那些人
select gender,group_concat(username) from userinfo group by gender; 

# 统计记录数量
select gender,group_concat(username) from userinfo group by gender with rollup;
-- 输出
+--------+--------------------------------------+
| gender | group_concat(username)               |
+--------+--------------------------------------+
| m      | Bob                                  |
| w      | admin,user,Alice,root,usr,client     |
| NULL   | Bob,admin,user,Alice,root,usr,client |
+--------+--------------------------------------+

分组过滤

# 先分组再过滤
select classid,group_concat(name) from userinfo group by classid having avg(height) > 160;

联合查询

# 笛卡尔积连接
-- 语法规则
SELECT <字段名> FROM <表1> CROSS JOIN <表2> [WHERE子句]
SELECT <字段名> FROM <表1>, <表2> [WHERE子句] 

-- 全部做笛卡尔积操作
SELECT * FROM tb_course CROSS JOIN tb_students_info;
-- 筛选满足条件的笛卡尔积（先做笛卡尔积再筛选）
SELECT * FROM tb_course CROSS JOIN tb_students_info WHERE tb_students_info.course_id = tb_course.id;

# 内敛查询
-- 语法规则
SELECT <字段名> FROM <表1> INNER JOIN <表2> [ON子句]
- 字段名：需要查询的字段名称。
- <表1><表2>：需要内连接的表名。
- INNER JOIN ：内连接中可以省略 INNER 关键字，只用关键字 JOIN。
- ON 子句：用来设置内连接的连接条件。

-- 直接内联
SELECT s.name,c.course_name FROM tb_students_info s INNER JOIN tb_course c ON s.course_id = c.id;
-- 过滤内联
SELECT
    s.name,
    c.course_name 
FROM
    tb_students_info s
    INNER JOIN tb_course c ON s.course_id = c.id 
    AND classid = 'php217';

# 外联查询
-- 左连接（左表全部显示
SELECT <字段名> FROM <表1> LEFT OUTER JOIN <表2> 
- 字段名：需要查询的字段名称。
- <表1><表2>：需要左连接的表名。
- LEFT OUTER JOIN：左连接中可以省略 OUTER 关键字，只使用关键字 LEFT JOIN。
- ON 子句：用来设置左连接的连接条件，不能省略。
-- 右连接（右表全部显示
SELECT <字段名> FROM <表1> RIGHT OUTER JOIN <表2> 
- 字段名：需要查询的字段名称。
- <表1><表2>：需要右连接的表名。
- RIGHT OUTER JOIN：右连接中可以省略 OUTER 关键字，只使用关键字 RIGHT JOIN。
- ON 子句：用来设置右连接的连接条件，不能省略。

删除

# 删除单个表中的数据
DELETE 
FROM
    userinfo 
WHERE
    gender = 'w' 
ORDER BY
    id 
    LIMIT 2
-- 语法规则
DELETE FROM <表名> [WHERE 子句] [ORDER BY 子句] [LIMIT 子句]

# 删除表中的全部数据（多条日志，不会重置参数）
delete from userinfo;

# 清空数据表(一条日志，会重置默认参数)
truncate userinfo;

#  根据条件删除表中指定数据
delete from userinfo where username = 'test';
